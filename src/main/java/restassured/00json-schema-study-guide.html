01

What is JSON Schema?

JSON Schema is a vocabulary that describes the shape, types, and constraints of JSON data. It is itself a JSON document. Think of it as a contract between producers and consumers of data.

Draft versions: Always declare $schema. The most widely used drafts are Draft-07, 2019-09, and 2020-12. Many validators still target Draft-07.
skeleton.json
{
  "$schema": "https://json-schema.org/draft/2020-12/schema",
  "$id":     "https://example.com/my-schema.json",
  "title":  "My Schema",
  "description": "What this schema validates",
  "type":  "object"
}

04

Complex Example — Employee Onboarding Payload

This schema validates a realistic HR payload with all major data types, nested objects, arrays, conditionals, and reusable definitions.

employee-schema.json
{
  "$schema":      "https://json-schema.org/draft/2020-12/schema",
  "$id":          "https://hr.example.com/schemas/employee.json",
  "title":        "Employee",
  "description":  "Schema for a new employee onboarding record",
  "type":         "object",

  // ── 1. Reusable Definitions ────────────────────────
  "$defs": {
    "address": {
      "type": "object",
      "required": ["street", "city", "country"],
      "properties": {
        "street":  { "type": "string", "minLength": 5  },
        "city":    { "type": "string"                   },
        "pincode": { "type": "string", "pattern": "^\\d{6}$" },
        "country": { "type": "string", "minLength": 2  }
      },
      "additionalProperties": false
    },

    "money": {
      "type":   "object",
      "required": ["amount", "currency"],
      "properties": {
        "amount":   { "type": "number", "minimum": 0, "multipleOf": 0.01 },
        "currency": { "type": "string", "enum": ["INR", "USD", "EUR"] }
      }
    }
  },

  // ── 2. Required top-level fields ───────────────────
  "required": [
    "employeeId", "fullName", "email",
    "department", "joiningDate", "salary"
  ],

  "properties": {

    // STRING  ── with pattern
    "employeeId": {
      "type":    "string",
      "pattern": "^EMP-[0-9]{4}$",
      "description": "Format: EMP-0001"
    },

    // STRING  ── length constraint
    "fullName": {
      "type":      "string",
      "minLength": 2,
      "maxLength": 100
    },

    // STRING  ── format keyword
    "email": {
      "type":   "string",
      "format": "email"
    },

    // INTEGER ── range
    "age": {
      "type":    "integer",
      "minimum": 18,
      "maximum": 65
    },

    // BOOLEAN
    "isRemote": {
      "type":    "boolean",
      "default": false
    },

    // STRING  ── enum (fixed set)
    "department": {
      "type": "string",
      "enum": ["Engineering", "QA", "HR", "Finance", "Product"]
    },

    // STRING  ── date format
    "joiningDate": {
      "type":   "string",
      "format": "date"
    },

    // NULL allowed  ── union type
    "terminationDate": {
      "type": ["string", "null"],
      "format": "date"
    },

    // OBJECT  ── $ref to reusable "money" definition
    "salary": {
      "$ref": "#/$defs/money"
    },

    // OBJECT  ── $ref to reusable "address" definition
    "address": {
      "$ref": "#/$defs/address"
    },

    // ARRAY   ── of strings (unique)
    "skills": {
      "type":        "array",
      "items": { "type": "string", "minLength": 2 },
      "minItems":    1,
      "maxItems":    20,
      "uniqueItems": true
    },

    // ARRAY   ── of objects
    "certifications": {
      "type": "array",
      "items": {
        "type": "object",
        "required": ["name", "issuedOn"],
        "properties": {
          "name":     { "type": "string"                   },
          "issuedOn": { "type": "string", "format": "date"    },
          "score":    { "type": "number", "minimum": 0, "maximum": 100 }
        }
      }
    },

    // CONDITIONAL ── if isRemote:true → requireEquipmentType
    "equipmentType": {
      "type": "string",
      "enum": ["laptop", "desktop", "none"]
    },

    // NUMBER  ── performance rating
    "performanceRating": {
      "type":             "number",
      "minimum":          1.0,
      "maximum":          5.0,
      "multipleOf":        0.5
    },

    // OBJECT  ── freeform metadata (any extra keys allowed)
    "metadata": {
      "type":                 "object",
      "additionalProperties": true
    }
  },

  // ── 3. Conditional: if remote, equipmentType required ──
  "if": {
    "properties": { "isRemote": { "const": true } }
  },
  "then": {
    "required": ["equipmentType"]
  },

  "additionalProperties": false
}

valid-payload.json
{
  "employeeId":  "EMP-0042",
  "fullName":    "Sumit Sagar",
  "email":       "sumit@example.com",
  "age":         28,
  "isRemote":    true,
  "department":  "QA",
  "joiningDate": "2024-03-01",
  "terminationDate": null,
  "salary": {
    "amount":   85000.00,
    "currency": "INR"
  },
  "address": {
    "street":  "MG Road",
    "city":    "Nagpur",
    "pincode": "440001",
    "country": "India"
  },
  "skills": ["Java", "Selenium", "RestAssured"],
  "certifications": [{
    "name":     "ISTQB",
    "issuedOn": "2023-06-15",
    "score":    82.5
  }],
  "equipmentType":      "laptop",
  "performanceRating":  4.5,
  "metadata": { "team": "X4V" }
}
employeeIdString matching regex ^EMP-[0-9]{4}$
ageInteger, range 18–65. Optional field.
isRemote: trueTriggers the if/then — makes equipmentType required.
terminationDateAccepts null via union type ["string","null"]
salaryReferences $defs/money. Amount must be multiple of 0.01.
skillsArray of strings, 1–20 items, all unique.
performanceRatingNumber 1.0–5.0, steps of 0.5 only.
additionalProperties: falseAny unknown key on root object will fail validation.

06

Schema Combiners

allOf
Value must be valid against all listed schemas. Used to merge/extend schemas.
anyOf
Value must be valid against at least one of the listed schemas.
oneOf
Value must be valid against exactly one schema. Mutual exclusion.
combiners.json
// allOf — must satisfy BOTH constraints
"salary": {
  "allOf": [
    { "type": "number", "minimum": 0 },
    { "multipleOf": 100 }
  ]
}

// oneOf — either a phone number OR an email, not both
"contact": {
  "oneOf": [
    { "type": "string", "pattern": "^\\+?[0-9]{10,13}$" },
    { "type": "string", "format": "email"                }
  ]
}

// if / then / else — conditional fields
"if": {   "properties": { "type": { "const": "contractor" } } },
"then": { "required": ["contractEndDate"]                  },
"else": { "required": ["permanentBand"]                   }
07

Reuse with $defs and $ref

Avoid repetition by defining sub-schemas once and referencing them anywhere. This is the JSON Schema equivalent of a shared model class.

reuse-example.json
{
  "$defs": {
    "PhoneNumber": {
      "type":    "string",
      "pattern": "^\\+?[0-9]{10,13}$"
    }
  },
  "properties": {
    "mobile":    { "$ref": "#/$defs/PhoneNumber" },
    "emergency": { "$ref": "#/$defs/PhoneNumber" },
    // $ref can also point to external schemas
    "bankAccount": {
      "$ref": "https://schemas.example.com/bank-account.json"
    }
  }
}
Note: In Draft-07, reusable definitions live under "definitions" (not $defs). The $defs key was introduced in Draft 2019-09.
